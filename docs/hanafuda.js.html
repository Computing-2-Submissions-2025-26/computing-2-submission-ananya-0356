import R from "./ramda.js";

/**
 * Shuffles a deck of hanafuda cards using Fisher-Yates algorithm.
 * This is the first step of the card game. This should happen
 * for when the cards are dealt to the players and the table.
 * Pure function returns new array, doesn't modify input.\
 * @example
 * const cards = [card1, card2, ..., card52];
 * const shuffled = shuffleDeck(cards);
 * @param {Array} cards - Array of card objects to shuffle
 * @returns {Array} New shuffled deck
 */
function shuffleDeck(cards) {
    const deck = [...cards];
    let i = deck.length - 1;
    let j;
    while (i > 0) {
        j = Math.floor(Math.random() * (i + 1));
        [deck[i], deck[j]] = [deck[j], deck[i]];
        i = i - 1;
    }
    return deck;
}

/**
 * Deals cards from shuffled deck to players and field.
 * Divides 48-card deck into 8 for each player
 * and 8 for the table, leaving 24 remaining deck.
 * @example
 * const shuffledDeck = shuffleDeck(allcards);
 * const game = dealCards(shuffledDeck);
 * @param {Array} deck - Shuffled deck of cards
 * @returns {Object} Contains player1Hand, player2Hand,
 * fieldCards, remainingDeck
 */
function dealCards(deck) {
    return {
        player1Hand: R.slice(0, 8, deck),
        player2Hand: R.slice(8, 16, deck),
        fieldCards: R.slice(16, 24, deck),
        remainingDeck: R.slice(24, Infinity, deck)
    };
}

/**
 * Checks what tag a card has been given from cards.js
 * This tag identifies the card to what Yakus it can
 * make up. e.g. "Red Poetry Slips"
 * @example
 * const isRedPoetrySlip = hasTag("red poetry slips", card);
 * // true
 * @param {string} tag - Tag to find
 * @param {Object} card - Card object
 * @returns {boolean} True if card has tag
 */
function hasTag(tag, card) {
    return R.includes(tag, R.defaultTo([], R.prop("tags", card)));
}

/**
 * Checks which of the five bright cards a player has.
 * This is because each bright card is unique and
 * different combos make different Yakus.
 * This is unlike the other category of cards.
 * @example
 * const hasMoon = hasBrightTag("moon", capturedCards);
 * // true
 * @param {string} tag - Tag to find
 * @param {Array} cards - Captured cards
 * @returns {boolean} True if has card with tag
 */
function hasBrightTag(tag, cards) {
    return R.any(function (card) {
        return hasTag(tag, card);
    }, cards);
}

/**
 * Calculates bright combination points.
 * These are point allocation for Yakus made up
 * of the Bright cards.
 * @example
 * const points = calculateBrightPoints(player.captured);
 * // 20 (All Five Brights)
 * @param {Array} cards - Captured cards
 * @returns {number} Points from bright combinations
 */
function calculateBrightPoints(cards) {
    const hasCrane = hasBrightTag("crane", cards);
    const hasMoon = hasBrightTag("moon", cards);
    const hasCurtain = hasBrightTag("curtain", cards);
    const hasPhoenix = hasBrightTag("phoenix", cards);
    const hasRain = hasBrightTag("rain", cards);

    if (hasCrane && hasMoon && hasCurtain && hasPhoenix && hasRain) {
        return 20;
    }
    if (hasMoon && hasCurtain && hasPhoenix && hasRain) {
        return 15;
    }
    if (hasMoon && hasCurtain && hasPhoenix) {
        return 12;
    }
    if (hasCrane && hasMoon) {
        return 10;
    }

    return 0;
}

/**
 * Filters through the player's captured cards and
 * counts how many belong to the Bright category.
 * @example

 * @param {Array} capturedCards - Cards the player has captured

function countBrights(capturedCards) {
    return R.length(R.filter(isBright, capturedCards));

/**
 * Checks if player has all three animal cards: boar, deer, butterfly.
 * Ino-Shika-Cho yakusu combination.
 * @example
 * const hasInoShikaCho = checkInoShikaCho(player.captured);
 * // true
 * @param {Array} cards - Captured cards to check
 * @returns {boolean} True if has all three animals
 */
function checkInoShikaCho(cards) {
    let hasBoar = false;
    let hasDeer = false;
    let hasButterfly = false;
    let i = 0;

    while (i < cards.length) {
        const card = cards[i];
        if (card.tags) {
            if (R.includes("boar", card.tags)) {
                hasBoar = true;
            }
            if (R.includes("deer", card.tags)) {
                hasDeer = true;
            }
            if (R.includes("butterfly", card.tags)) {
                hasButterfly = true;
            }
        }
        i = i + 1;
    }

    return hasBoar && hasDeer && hasButterfly;
}

/**
 * Calculates total score from captured cards.
 * @example
 * const score = calculateScore(player.captured);
 * // 13
 * @param {Array} cards - Player's captured cards
 * @returns {number} Total points scored
 */
function calculateScore(cards) {
    let totalPoints = 0;

    let brightPoints = calculateBrightPoints(cards);
    totalPoints = totalPoints + brightPoints;

    let redCount = countRedPoetrySlips(cards);
    if (redCount >= 3) {
        totalPoints = totalPoints + 3 + (redCount - 3);
    }

    let blueCount = countBluePoetrySlips(cards);
    if (blueCount >= 3) {
        totalPoints = totalPoints + 3 + (blueCount - 3);
    }

    let poetryCount = countPoetrySlips(cards);
    if (poetryCount >= 5) {
        totalPoints = totalPoints + 5 + (poetryCount - 5);
    }

    let seedCount = countSeeds(cards);
    if (seedCount >= 5) {
        totalPoints = totalPoints + 5 + (seedCount - 5);
    }

    if (checkInoShikaCho(cards)) {
        totalPoints = totalPoints + 5;
    }

    let chaffCount = countChaff(cards);
    if (chaffCount >= 10) {
        totalPoints = totalPoints + 1;
    }

    return totalPoints;
}

/**
 * Looks through the cards on the table to find
 * one with the same month as the player's card.
 * @example
 * const match = findMatchingFieldCard(handCard, fieldCards);
 * // Returns the matching field card, or null
 *    if no match exists.
 * @param {Object} handCard - Player's card to match
 * @param {Array} fieldCards - Cards currently on field
 * @returns {Object|null} Matching field card or null
 */
function findMatchingFieldCard(handCard, fieldCards) {
    const handMonth = R.prop("month", handCard);
    const predicate = function (card) {
        return R.propEq("month", handMonth, card);
    };
    return R.find(predicate, fieldCards) || null;
}

/**
 * Removes a specific card from a group of cards, by
 * using its unique ID. It returns a new array that
 * does not contain that card.
 * @example
 * const updatedHand = removeCardById(selectedCard.id, player.hand);
 * @param {string} cardId - Card id to remove
 * @param {Array} cards - Array to remove from
 * @returns {Array} New array without card
 */
function removeCardById(cardId, cards) {
    const predicate = function (card) {
        return !R.propEq("id", cardId, card);
    };
    return R.filter(predicate, cards);
}

/**
 * Draws top card from deck.
 * Returns first card or null if deck is empty.
 * @example
 * const drawnCard = drawDeckCard(gamestate.deck);
 * @param {Array} deck - Current deck
 * @returns {Object|null} Top card or null
 */
function drawDeckCard(deck) {
    if (R.isEmpty(deck)) {
        return null;
    }
    return R.head(deck);
}

/**
 * Identifies all Yaku completed by the player and
 * returns their names alongside the points awarded.
 * @example
 * const completedYaku = getCompletedYaku(player.captured);
 * // [
 * //   { name: "Moon Viewing", points: 10 },
 * //   { name: "Red Poetry Slips", points: 3 }
 * // ]
 * @param {Array} cards - Player's captured cards
 * @returns {Array} Array of {name, points} objects
 */
function getCompletedYaku(cards) {
    let yaku = [];

    let brightPoints = calculateBrightPoints(cards);
    if (brightPoints > 0) {
        let brightName = "";
        if (brightPoints === 20) {
            brightName = "All Five Brights";
        } else if (brightPoints === 15) {
            brightName = "Four Brights";
        } else if (brightPoints === 12) {
            brightName = "Three Lights";
        } else if (brightPoints === 10) {
            brightName = "Moon Viewing";
        }

        yaku = R.append(
            {
                name: brightName,
                points: brightPoints
            },
            yaku
        );
    }

    let redCount = countRedPoetrySlips(cards);
    if (redCount >= 3) {
        yaku = R.append(
            {
                name: "Red Poetry Slips",
                points: 3 + (redCount - 3)
            },
            yaku
        );
    }

    let blueCount = countBluePoetrySlips(cards);
    if (blueCount >= 3) {
        yaku = R.append(
            {
                name: "Blue Poetry Slips",
                points: 3 + (blueCount - 3)
            },
            yaku
        );
    }

    let poetryCount = countPoetrySlips(cards);
    if (poetryCount >= 5) {
        yaku = R.append(
            {
                name: "Poetry Slips",
                points: 5 + (poetryCount - 5)
            },
            yaku
        );
    }

    let seedCount = countSeeds(cards);
    if (seedCount >= 5) {
        yaku = R.append(
            {
                name: "Seeds",
                points: 5 + (seedCount - 5)
            },
            yaku
        );
    }

    if (checkInoShikaCho(cards)) {
        yaku = R.append(
            {
                name: "Ino-Shika-Cho",
                points: 5
            },
            yaku
        );
    }

    let chaffCount = countChaff(cards);
    if (chaffCount >= 10) {
        yaku = R.append(
            {
                name: "Chaff",
                points: 1
            },
            yaku
        );
    }

    return yaku;
}
